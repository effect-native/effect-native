## Implementation Notes

### API Summary
```typescript
// Screen object wraps a Terminal for assertions
class Screen {
  static fromTerminal(terminal: Terminal): Screen
  text(): string           // Plain text content (strips ANSI)
  ansi(): string          // Content with ANSI escape sequences preserved
  getCell(row: number, col: number): CellInfo | undefined
  getCursor(): { row: number; col: number }
}

// Polling helper
function waitFor<T>(
  predicate: () => T | Promise<T>,
  options?: { timeout?: number; interval?: number }
): Promise<NonNullable<T>>

// Error for timeout
class WaitForTimeoutError extends Error
```
